int partition(int arr[],int l,int h)

{

                    int i=l,j=h;

                    int pivot=arr[l];

                    while(i<j)

                    {

                    while(arr[i]<pivot)

                    {

                                         i++;

                    }

                    while(arr[j]>pivot)

                    {

                                         j--;

                    }

                    int temp;

                    temp=arr[i];

                    arr[i]=arr[j];

                    arr[j]=temp;

    }

    return j;
